(defn gcd [a b]
  (if (zero? b)
    a
    (recur b (mod a b))))

(def ratmap
  (zipmap
   (for [numer (range 1 101)
         denom (range 1 101)]
     (round (/ numer denom)))
   (for [numer (range 1 101)
         denom (range 1 101)]
     [numer denom])))

    [(/ (first (ratmap (round decimal))) divisor)
